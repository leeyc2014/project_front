package com.barcode.service;

import com.barcode.domain.LogisError;
import com.barcode.domain.LogisLog;
import com.barcode.domain.LogisMove;
import com.barcode.dto.DateCountResponse;
import com.barcode.dto.PagedResponse;
import com.barcode.repository.LogisErrorRepository;
import com.barcode.repository.LogisLogRepository;
import com.barcode.repository.LogisMoveRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.List;

/**
 * LogisLog, LogisMove, LogisError 조회 서비스 (날짜별 개수, 날짜별 페이지 조회)
 */
@Service
@RequiredArgsConstructor
public class LogisQueryService {

    private static final int DEFAULT_PAGE_SIZE = 20;

    private final LogisLogRepository logisLogRepository;
    private final LogisMoveRepository logisMoveRepository;
    private final LogisErrorRepository logisErrorRepository;

    public DateCountResponse countLogisLogByDate(LocalDate date) {
        long count = logisLogRepository.countByEventTimeDate(date);
        return new DateCountResponse(date.toString(), count);
    }

    public PagedResponse<LogisLog> findLogisLogByDate(LocalDate date, int page) {
        Pageable pageable = PageRequest.of(page, DEFAULT_PAGE_SIZE);
        var springPage = logisLogRepository.findByEventTimeDate(date, pageable);
        return toPagedResponse(springPage.getContent(), springPage);
    }

    public DateCountResponse countLogisMoveByDate(LocalDate date) {
        long count = logisMoveRepository.countByEventTimeDate(date);
        return new DateCountResponse(date.toString(), count);
    }

    public PagedResponse<LogisMove> findLogisMoveByDate(LocalDate date, int page) {
        Pageable pageable = PageRequest.of(page, DEFAULT_PAGE_SIZE);
        var springPage = logisMoveRepository.findByEventTimeDate(date, pageable);
        return toPagedResponse(springPage.getContent(), springPage);
    }

    public DateCountResponse countLogisErrorByDate(LocalDate date) {
        long count = logisErrorRepository.countByEventTimeDate(date);
        return new DateCountResponse(date.toString(), count);
    }

    public PagedResponse<LogisError> findLogisErrorByDate(LocalDate date, int page) {
        Pageable pageable = PageRequest.of(page, DEFAULT_PAGE_SIZE);
        var springPage = logisErrorRepository.findByEventTimeDate(date, pageable);
        return toPagedResponse(springPage.getContent(), springPage);
    }

    private <T> PagedResponse<T> toPagedResponse(List<T> content, org.springframework.data.domain.Page<?> springPage) {
        return new PagedResponse<>(
                content,
                springPage.getNumber(),
                springPage.getSize(),
                springPage.getTotalElements(),
                springPage.getTotalPages(),
                springPage.isFirst(),
                springPage.isLast()
        );
    }
}

package com.barcode.repository;

import com.barcode.domain.LogisLog;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.stereotype.Repository;

import java.time.LocalDate;

@Repository
public interface LogisLogRepository extends JpaRepository<LogisLog, Long> {

    @Query("SELECT COUNT(l) FROM LogisLog l WHERE CAST(l.eventTime AS date) = :targetDate")
    long countByEventTimeDate(@Param("targetDate") LocalDate date);

    @Query("SELECT l FROM LogisLog l WHERE CAST(l.eventTime AS date) = :targetDate ORDER BY l.eventTime ASC, l.id ASC")
    Page<LogisLog> findByEventTimeDate(@Param("targetDate") LocalDate date, Pageable pageable);
}

package com.barcode.controller;

import com.barcode.dto.LogisLogCsvUploadResponse;
import com.barcode.dto.LogisLogRequest;
import com.barcode.dto.LogisLogResponse;
import com.barcode.service.LogisLogService;
import lombok.RequiredArgsConstructor;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.multipart.MultipartFile;

/**
 * LogisLog 데이터 처리 컨트롤러
 */
@RestController
@RequestMapping("/api/v1")
@RequiredArgsConstructor
public class LogisLogController {

    private final LogisLogService logisLogService;

    /**
     * LogisLog 데이터를 받아서 저장하고, 검증 통과 시 LogisMove에도 추가합니다.
     * 
     * @param request LogisLog 요청 데이터
     * @return 처리 결과 응답
     */
    @PostMapping("/logis-log")
    public ResponseEntity<LogisLogResponse> createLogisLog(@RequestBody LogisLogRequest request) {
        LogisLogResponse response = logisLogService.processLogisLog(request);
        
        if (response.isSuccess() && response.getLogisMoveId() != null) {
            return ResponseEntity.ok(response);
        } else if (response.isSuccess()) {
            // LogisLog는 저장되었지만 LogisMove에는 추가되지 않은 경우
            return ResponseEntity.ok(response);
        } else {
            return ResponseEntity.badRequest().body(response);
        }
    }

    /**
     * CSV 파일을 업로드하여 LogisLog에 일괄 저장합니다.
     * 탭 또는 쉼표 구분, 첫 줄은 헤더로 제외됩니다.
     * 각 행은 검증 후 통과 시 LogisMove에도 추가됩니다.
     *
     * @param file CSV 파일 (multipart/form-data, key: file)
     * @return 업로드 결과 (총 행 수, 저장 수, LogisMove 추가 수, 오류 수)
     */
    @PostMapping(value = "/logis-log/upload", consumes = "multipart/form-data")
    public ResponseEntity<LogisLogCsvUploadResponse> uploadLogisLogCsv(
            @RequestParam("file") MultipartFile file) {
        LogisLogCsvUploadResponse response = logisLogService.processLogisLogCsv(file);
        if (!response.isSuccess() && response.getTotalRows() == 0) {
            return ResponseEntity.badRequest().body(response);
        }
        return ResponseEntity.ok(response);
    }
}
